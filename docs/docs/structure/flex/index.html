Contain

You can set .contain sizes and spacing inside the .less file.


<!-- contain inside responsive sizes -->
<div class="contain">
  <!-- content -->
</div>

<!-- fluid size contain -->
<div class="contain fluid">
  <!-- content -->
</div>
